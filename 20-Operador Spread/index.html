20-Operador Spread

Usamos (...numeros), para poder ultilizar os valores do primeiro array

let numeros = [1, 2, 3, 4]; let outros = [...numeros,5,6,7,8]; console.log(outros);

Outro exemplo: Com o obejto, incluindo cidade e estado.

let info = { nome: "Aironys", sobrenome: "Garrido", idade: 31 }; console.log(info); let info2 = { ...info, cidade:"Campina Granda", estado: "Paraíba", pais: "Brasil" }; console.log(info2);

Agora com funções

function adcionar(info) { return novaInfo = { ...info, status:0, token:"1651651651651", data: "25/10/2022" } return novaInfo; } console.log( adcionar({ nome: "Aironys ANS", sobrenome: "Garrido", idade: 31, estado: "Paraíba", cidade: "Serraria", cargo: "Desenvolvedor Web" }));
